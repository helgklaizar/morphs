import { useState, useEffect } from 'react';
 
const blueprints = [
  { id: 'restaurant', icon: '🍽️', title: 'Restaurant', desc: 'Для кафе и ресторанов' },
  { id: 'retail', icon: '🛍️', title: 'Retail', desc: 'Магазины и склады' },
  { id: 'service', icon: '⚙️', title: 'Service', desc: 'Оказание услуг' }
];
 
const checklistItems = [
  { id: 'pos', title: 'POS Оффлайн Терминал', desc: 'Автономная работа кассира', required: true },
  { id: 'sync', title: 'Локальная SQLite с Синхронизацией', desc: 'Хранение данных на устройстве с P2P пушем', required: true },
  { id: 'crm', title: 'Умная CRM', desc: 'Лояльность и анализ поведения клиентов', required: false },
  { id: 'hr', title: 'Управление Сотрудниками', desc: 'Графики и KPI', required: false },
];
 
export default function App() {
  const [selectedBusiness, setSelectedBusiness] = useState('restaurant');
  const [selectedModules, setSelectedModules] = useState<string[]>(['pos', 'sync', 'crm']);
  const [profile, setProfile] = useState<{name: string, tier: string} | null>(null);
 
  useEffect(() => {
    fetch("http://localhost:8000/api/v1/profile")
      .then(res => res.json())
      .then(data => setProfile(data))
      .catch(e => console.error("API error", e));
  }, []);
 
  const toggleModule = (id: string) => {
    if (selectedModules.includes(id)) {
      setSelectedModules(selectedModules.filter(m => m !== id));
    } else {
      setSelectedModules([...selectedModules, id]);
    }
  };
 
  const [isGenerating, setIsGenerating] = useState(false);
 
  const startGeneration = async () => {
    setIsGenerating(true);
    try {
      await fetch("http://localhost:8000/api/v1/generate", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ business_type: selectedBusiness, modules: selectedModules })
      });
      // ИИ пишет код в фоне (2-3 минуты). Tauri обновится через HMR, 
      // когда файл появится на диске в компонентах.
    } catch (e) {
      console.error("Core Mind is down:", e);
      setIsGenerating(false);
    }
  };
 
  return (
    <div className="min-h-screen bg-[#0B0D17] text-white py-16 px-6 sm:px-12 flex flex-col items-center justify-center overflow-x-hidden">
      <div className="w-full max-w-4xl glass-panel p-8 md:p-12 relative overflow-hidden transition-all duration-500">
        <div className="absolute -top-32 -right-32 w-64 h-64 bg-primary blur-[120px] opacity-20 pointer-events-none rounded-full" />
        
        {isGenerating ? (
          <div className="flex flex-col items-center justify-center h-64 space-y-6 relative z-10 text-center">
             <div className="w-16 h-16 border-4 border-surface border-t-primary rounded-full animate-spin"></div>
             <h2 className="text-2xl font-bold text-white">ИИ-Архитектор Моделирует Ядро...</h2>
             <p className="text-textMuted max-w-md">Llama-3 вычисляет топологию бизнес-процессов на базе '{selectedBusiness}' и пишет React-компоненты. <br/>Tauri окно автоматически обновится (HMR).</p>
          </div>
        ) : (
          <>
            <header className="mb-12 text-center relative z-10">
              <div className="absolute right-0 top-0 inline-flex items-center space-x-2 px-3 py-1 rounded-full border border-gray-700 bg-white/5">
                <span className="text-gray-300 text-xs font-semibold uppercase">{profile ? `${profile.name} - ${profile.tier} TIER` : 'Offline'}</span>
              </div>
              <div className="inline-flex items-center space-x-2 px-3 py-1 rounded-full border border-primary/30 bg-primary/10 mb-6 mt-8 md:mt-0">
                <span className="w-2 h-2 rounded-full bg-primary animate-pulse" />
                <span className="text-primary text-xs font-semibold tracking-widest uppercase">Morphs Setup Wizard</span>
              </div>
              <h1 className="text-4xl md:text-5xl font-extrabold tracking-tight mb-4">Настрой своё <span className="text-transparent bg-clip-text bg-gradient-to-r from-primary to-orange-400">ИИ-Ядро</span></h1>
              <p className="text-textMuted text-lg max-w-xl mx-auto">Выбери профиль бизнеса. Конфигуратор сгенерирует локальный комбайн и напишет код интерфейса под тебя.</p>
            </header>
 
            {/* Business Selector */}
            <div className="mb-10 relative z-10">
              <h2 className="text-sm uppercase tracking-widest text-textMuted font-semibold mb-4">Тип бизнеса</h2>
              <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
                {blueprints.map(bp => (
                  <button
                    key={bp.id}
                    onClick={() => setSelectedBusiness(bp.id)}
                    className={`p-6 rounded-xl border text-left transition-all duration-300 flex flex-col items-center justify-center space-y-3
                      ${selectedBusiness === bp.id 
                        ? 'border-primary bg-primary/5 shadow-[0_0_30px_rgba(255,107,0,0.15)] ring-1 ring-primary/50' 
                        : 'border-borderBase hover:border-gray-600 bg-surface/50'}`}
                  >
                    <div className="text-4xl">{bp.icon}</div>
                    <div className="font-medium text-lg text-white">{bp.title}</div>
                    <div className="text-xs text-textMuted text-center">{bp.desc}</div>
                  </button>
                ))}
              </div>
            </div>
 
            {/* Checklist */}
            <div className="mb-12 relative z-10 bg-black/20 p-6 rounded-2xl border border-borderBase">
              <div className="flex items-center justify-between mb-6">
                <h2 className="text-sm uppercase tracking-widest text-textMuted font-semibold">Наполнение Модулями (Skills)</h2>
                <span className="text-xs text-primary bg-primary/10 px-2 py-1 rounded-lg">Кастомизируемо</span>
              </div>
              
              <div className="space-y-3">
                {checklistItems.map(item => (
                  <label key={item.id} className="flex items-center p-4 rounded-xl border border-borderBase hover:bg-white/5 transition-colors cursor-pointer group">
                    <input 
                      type="checkbox"
                      checked={selectedModules.includes(item.id)}
                      onChange={() => !item.required && toggleModule(item.id)}
                      disabled={item.required}
                      className="w-5 h-5 rounded border-gray-600 text-primary focus:ring-primary focus:ring-offset-0 bg-transparent disabled:opacity-50"
                    />
                    <div className="ml-4 flex-1">
                      <div className="text-white font-medium flex items-center">
                        {item.title}
                        {item.required && <span className="ml-2 text-[10px] uppercase text-primary border border-primary/30 px-1.5 py-0.5 rounded">Core</span>}
                      </div>
                      <div className="text-sm text-textMuted">{item.desc}</div>
                    </div>
                  </label>
                ))}
              </div>
            </div>
 
            {/* Action Button */}
            <div className="flex justify-end relative z-10">
              <button 
                onClick={startGeneration}
                className="px-8 py-4 bg-primary text-white font-semibold rounded-xl hover:bg-orange-500 hover:shadow-[0_0_30px_rgba(255,107,0,0.4)] transition-all duration-300 transform hover:-translate-y-1">
                Генерировать (Llama-3) →
              </button>
            </div>
          </>
        )}
      </div>
    </div>
  );
}
